2. Create `src/components/layout/BottomNav.tsx` (per D-04 bottom tab bar, D-05 tabs Home + History):
   ```typescript
   import { NavLink } from 'react-router';

   export default function BottomNav() {
     const linkClass = ({ isActive }: { isActive: boolean }) =>
       `flex flex-col items-center justify-center min-h-[44px] min-w-[44px] px-4 py-2 text-xs font-medium transition-colors ${
         isActive ? 'text-accent' : 'text-text-secondary'
       }`;

     return (
       <nav className="flex justify-around items-center border-t border-border bg-bg-secondary pb-[env(safe-area-inset-bottom)]">
         <NavLink to="/" end className={linkClass}>
           <span className="text-lg mb-0.5">&#x1F3E0;</span>
           Home
         </NavLink>
         <NavLink to="/history" className={linkClass}>
           <span className="text-lg mb-0.5">&#x1F4CB;</span>
           History
         </NavLink>
       </nav>
     );
   }
   ```
   NOTE: Use simple Unicode icons initially (house U+1F3E0, clipboard U+1F4CB). The `end` prop on the Home NavLink prevents it from matching all routes. Per Research recommendation, the "Active Workout" tab will be added conditionally in Phase 2 when workout state exists.

3. Create `src/components/layout/AppLayout.tsx` (per D-01 dark theme, PLT-01 safe area insets):
   ```typescript
   import { Outlet } from 'react-router';
   import BottomNav from './BottomNav';

   export default function AppLayout() {
     return (
       <div className="dark flex flex-col h-dvh bg-bg-primary text-text-primary">
         <div className="pt-[env(safe-area-inset-top)]" />
         <main className="flex-1 overflow-y-auto px-4 pb-4">
           <Outlet />
         </main>
         <BottomNav />
       </div>
     );
   }
   ```
   Key classes: `h-dvh` (not `h-screen`) to avoid iOS Safari 100vh bug. `dark` class on root div to activate dark variant. `overflow-y-auto` on main for scrollable content. `pb-4` for bottom spacing before nav.
